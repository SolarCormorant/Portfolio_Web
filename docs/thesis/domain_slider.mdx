import React, { useState } from 'react';
import DomainSlider from '@site/src/components/DomainSlider';

export default function DomainSliderDemoPage() {
  const [range, setRange] = useState([20, 80]);

  return (
    <>
      <h1>Domain Slider Demo</h1>
      <DomainSlider
        min={0}
        max={100}
        step={1}
        label="Aralık Seçin"
        onChange={(vals) => setRange(vals)}
      />
      <p>Seçilen aralık: {range[0]} - {range[1]}</p>
    </>
  );
}
